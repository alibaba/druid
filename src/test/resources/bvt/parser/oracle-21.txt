	FROM (

		FROM (
			SELECT t.*, rct.parent_id, RCT.NAME AS trunk_name
			FROM x_1_2 t, k_2_3 rct

		) tt

		CONNECT BY PRIOR parent_id = trunk_id
		ORDER BY load_order NULLS LAST
