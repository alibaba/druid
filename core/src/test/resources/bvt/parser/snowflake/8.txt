--------------------
SELECT id, name
	, (
		SELECT COUNT(*)
		FROM orders o
		WHERE o.customer_id = c.id
	) AS order_count
FROM customers c

--------------------
SELECT DATE_TRUNC('month', created_at) AS month, COUNT(*)
FROM orders
GROUP BY month
ORDER BY month

	, COUNT(*)
