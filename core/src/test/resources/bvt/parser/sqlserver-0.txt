SELECT TOP 5 *
FROM (
    SELECT ROW_NUMBER() OVER (ORDER BY RAND()) PAGE_ROW_NUMBER, *
    FROM (
        SELECT tc.* FROM TC_CarIdentify AS tc WHERE 1 = 1
    ) AS PAGE_TABLE_ALIAS
) AS PAGE_TABLE_ALIAS
WHERE PAGE_ROW_NUMBER > 0 ORDER BY PAGE_ROW_NUMBER
