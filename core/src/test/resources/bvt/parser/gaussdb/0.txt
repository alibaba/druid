CREATE UNLOGGED TABLE lineitem (
	L_ORDERKEY BIGINT NOT NULL,
	L_QUANTITY DECIMAL(15, 2) NOT NULL,
	L_SHIPDATE DATE NOT NULL,
   PARTIAL CLUSTER KEY  (l_shipdate, l_quantity)
)
WITH (
	'orientation' = column
)
DISTRIBUTE BY hash (L_ORDERKEY);
